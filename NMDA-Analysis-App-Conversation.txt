# ---- Helpers ----
fmt_p <- function(p) {
  if (is.na(p)) return(NA_character_)
  if (p < 0.001) return("<0.001")
  sprintf("%.3f", p)
}
cont_summary <- function(data, var, group = "D-NMDA") {
  x0 <- data %>% filter(.data[[group]] == 0) %>% pull(.data[[var]]) %>% na.omit()
  x1 <- data %>% filter(.data[[group]] == 1) %>% pull(.data[[var]]) %>% na.omit()
  
  # Shapiro-Wilk normality (only valid if n >= 3)
  p0 <- if (length(x0) >= 3) shapiro.test(x0)$p.value else 0
  p1 <- if (length(x1) >= 3) shapiro.test(x1)$p.value else 0
  normal <- (p0 > 0.05) & (p1 > 0.05)
  
  if (normal) {
    # Welch t-test
    tt <- t.test(x0, x1, var.equal = FALSE)
    out0 <- sprintf("%.2f \u00B1 %.2f", mean(x0), sd(x0))
    out1 <- sprintf("%.2f \u00B1 %.2f", mean(x1), sd(x1))
    test <- "Welch t-test"
    p <- tt$p.value
  } else {
    # Wilcoxon rank-sum (Mann–Whitney U)
    wt <- wilcox.test(x0, x1, exact = FALSE)
    out0 <- sprintf("%.2f (IQR %.2f\u2013%.2f)",
                    median(x0), quantile(x0, 0.25), quantile(x0, 0.75))
    out1 <- sprintf("%.2f (IQR %.2f\u2013%.2f)",
                    median(x1), quantile(x1, 0.25), quantile(x1, 0.75))
    test <- "Mann\u2013Whitney U (Wilcoxon rank-sum)"
    p <- wt$p.value
  }
  
  tibble(
    Variable = var,
    `No D-NMDA` = out0,
    `D-NMDA` = out1,
    `p-value` = fmt_p(p),
    `Statistical test` = test
  )
}
cat_summary <- function(data, var, group = "D-NMDA") {
  tab <- table(data[[var]], data[[group]], useNA = "ifany")
  
  # Default: chi-square
  chi <- suppressWarnings(chisq.test(tab, correct = FALSE))
  expected <- chi$expected
  
  # Fisher only when 2x2 AND any expected cell < 5
  use_fisher <- (nrow(tab) == 2 && ncol(tab) == 2 && any(expected < 5))
  
  if (use_fisher) {
    p <- fisher.test(tab)$p.value
    test <- "Fisher\u2019s exact"
  } else {
    p <- chi$p.value
    test <- "Chi-square"
  }
  
  # Build rows for each level
  counts0 <- tab[, "0"]
  counts1 <- tab[, "1"]
  n0 <- sum(counts0, na.rm = TRUE)
  n1 <- sum(counts1, na.rm = TRUE)
  
  out <- tibble(
    Level = rownames(tab),
    `No D-NMDA` = sprintf("%d (%.1f%%)", counts0, 100 * counts0 / n0),
    `D-NMDA` = sprintf("%d (%.1f%%)", counts1, 100 * counts1 / n1),
    `p-value` = fmt_p(p),
    `Statistical test` = test
  ) %>%
    mutate(Variable = paste0(var, ": ", Level)) %>%
    select(Variable, `No D-NMDA`, `D-NMDA`, `p-value`, `Statistical test`)
  
  out
}

# ---- Helper: Linear regression ----
run_linear <- function(data, outcome) {
  formula_str <- paste0("`", outcome, "` ~ `D-NMDA` + ",
                        paste(covariates, collapse = " + "))
  fit <- lm(as.formula(formula_str), data = data)
  
  tidy(fit, conf.int = TRUE) %>%
    filter(term == "`D-NMDA`1") %>%
    mutate(
      Outcome         = outcome,
      Model           = "Linear regression",
      Estimate_label  = sprintf("β = %.3f (95%% CI: %.3f, %.3f)",
                                estimate, conf.low, conf.high),
      p_fmt           = ifelse(p.value < 0.001, "<0.001", sprintf("%.3f", p.value))
    ) %>%
    select(Outcome, Model, Estimate_label, p_fmt)
}

# Minimum Detectable Difference — with Figure
# =========================
library(pwr)
library(ggplot2)
library(dplyr)
# ---- Parameters from your data ----
sd0_mme    <- 179.38
sd1_mme    <- 146.87
n0_mme     <- 70
n1_mme     <- 23
mean0_mme  <- 363.07

pooled_sd <- sqrt(((n0_mme - 1) * sd0_mme^2 + (n1_mme - 1) * sd1_mme^2) /
                    (n0_mme + n1_mme - 2))
# ---- MDD at 80% and 90% power ----
mdd_80 <- pwr.t2n.test(n1 = n0_mme, n2 = n1_mme,
                       sig.level = 0.05, power = 0.80)$d * pooled_sd
mdd_90 <- pwr.t2n.test(n1 = n0_mme, n2 = n1_mme,
                       sig.level = 0.05, power = 0.90)$d * pooled_sd
observed_diff <- abs(mean0_mme - mean1_mme)
cat(sprintf("Pooled SD:               %.2f MME\n", pooled_sd))
cat(sprintf("MDD at 80%% power:        %.1f MME\n", mdd_80))
cat(sprintf("MDD at 90%% power:        %.1f MME\n", mdd_90))
cat(sprintf("Observed difference:     %.1f MME\n", observed_diff))
# ---- Power curve across a range of MME differences ----
diff_seq <- seq(10, 300, by = 1)
power_by_diff <- tibble(
  mme_diff = diff_seq,
  cohens_d = diff_seq / pooled_sd,
  power    = mapply(function(d) {
    pwr.t2n.test(n1 = n0_mme, n2 = n1_mme,
                 d = d, sig.level = 0.05)$power
  }, cohens_d)
)
# ---- Figure ----
fig_mdd <- ggplot(power_by_diff, aes(x = mme_diff, y = power)) +
  geom_line(color = "#2166AC", linewidth = 1.2) +

  fit_primary <- lm(as.formula(paste0("`", v, "` ~ `D-NMDA` + ",

  fit_pp      <- lm(as.formula(paste0("`", v, "` ~ `D-NMDA` + ",

  tibble(
    Outcome         = v,
    `β primary`     = round(b_primary, 3),
    `β per-protocol`= round(b_pp, 3),
    `% change`      = round(pct_chg, 1),
    Stable          = ifelse(pct_chg <= 20, "Yes", "No")
  )
})

Yes — it's deliberately built to be very tolerant. Here's the picture:

**Procedure specifically:** Not used by any analysis at all (Table 1, Table 2, regression, and power figure don't reference it). You can delete it from the Excel and the app won't notice.

**Auto-detection adapts on every upload:** The "what's continuous vs categorical" pass runs on whatever columns are actually present, so a slimmer dataset just produces a slimmer set of options in the Variables tab dropdowns.

**Multivariate covariates and MDD outcome:** Same story — if you delete `Total_MME`, the power-curve dropdown defaults to whatever's first in your continuous list (you can change it in the Variables tab). **The one thing you must keep:** the group column (`D-NMDA` by default — the binary 0/1 column defining the two cohorts). If that's missing the app stops with a clear red error explaining the problem and listing what columns it did find. So in practice, residents can drop columns, rename columns, or add brand-new columns to their Excel files and the app will just adapt. The only "rigid" requirement is one binary group column.
